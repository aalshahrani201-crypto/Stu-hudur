function hideLoginScreen(user) { const screen = document.getElementById('loginScreen'); screen.style.transition = 'opacity .5s, transform .5s'; screen.style.opacity = '0'; screen.style.transform = 'scale(1.05)'; setTimeout(() => { screen.style.display = 'none'; updateTopbarUser(user); // Hide admin features for non-admins if (user.role !== 'admin') { const mgBtn = document.getElementById('manageUsersBtn'); if (mgBtn) mgBtn.style.display = 'none'; } showToast(`أهلاً ${user.fullName} 👋`, '✅'); }, 500); } function updateTopbarUser(user) { document.getElementById('topUserAvatar').textContent = user.fullName[0]; document.getElementById('topUserName').textContent = user.fullName; const roles = { admin: '🔴 مدير النظام', teacher: '🟡 معلم', viewer: '🟢 مشاهد' }; document.getElementById('topUserRole').textContent = roles[user.role] || user.role; } function doLogout() { localStorage.removeItem('rememberSession'); Auth.currentUser = null; saveAuthDB(Auth); const screen = document.getElementById('loginScreen'); screen.style.display = 'flex'; screen.style.opacity = '0'; screen.style.transform = 'scale(1.05)'; setTimeout(() => { screen.style.transition = 'opacity .4s, transform .4s'; screen.style.opacity = '1'; screen.style.transform = 'scale(1)'; }, 10); document.getElementById('loginUser').value = ''; document.getElementById('loginPass').value = ''; document.getElementById('loginError').classList.remove('show'); document.getElementById('attemptsInfo').textContent = ''; document.getElementById('loginForm').style.display = 'block'; document.getElementById('lockedForm').style.display = 'none'; toggleUserDropdown(true); }
